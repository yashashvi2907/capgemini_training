/*

Create Two Tables for JOIN Practice
=========================================================
*/

CREATE TABLE departments(
dept_id INT PRIMARY KEY,
dept_name VARCHAR(50)
);


CREATE TABLE employees(
emp_id INT PRIMARY KEY,
name VARCHAR(50),
salary INT,
dept_id INT,
manager_id INT
);

/*
=========================================================
QUESTION 1
Display employee name with department name
=========================================================
*/

SELECT e.name, d.dept_name
FROM employees e
INNER JOIN departments d
ON e.dept_id = d.dept_id;



/*
=========================================================
QUESTION 2
Display employee name, salary and department name
=========================================================
*/

SELECT e.name, e.salary, d.dept_name
FROM employees e
INNER JOIN departments d
ON e.dept_id = d.dept_id;



/*
=========================================================
QUESTION 3
Find employees working in IT department
=========================================================
*/

SELECT e.name
FROM employees e
INNER JOIN departments d
ON e.dept_id = d.dept_id
WHERE d.dept_name = 'IT';



/*
=========================================================
QUESTION 4
Display all employees even if department does not exist
LEFT JOIN
=========================================================
*/

SELECT e.name, d.dept_name
FROM employees e
LEFT JOIN departments d
ON e.dept_id = d.dept_id;



/*
=========================================================
QUESTION 5
Display all departments even if no employee works there
RIGHT JOIN
=========================================================
*/

SELECT e.name, d.dept_name
FROM employees e
RIGHT JOIN departments d
ON e.dept_id = d.dept_id;



/*
=========================================================
QUESTION 6
Find employees who do not belong to any department
=========================================================
*/

SELECT e.name
FROM employees e
LEFT JOIN departments d
ON e.dept_id = d.dept_id
WHERE d.dept_id IS NULL;



/*
=========================================================
QUESTION 7
Display employees and their managers (SELF JOIN)
=========================================================
*/

SELECT 
e.name AS Employee,
m.name AS Manager
FROM employees e
LEFT JOIN employees m
ON e.manager_id = m.emp_id;



/*
=========================================================
QUESTION 8
Find employees whose manager is Shalini
=========================================================
*/

SELECT e.name
FROM employees e
JOIN employees m
ON e.manager_id = m.emp_id
WHERE m.name = 'Shalini';



/*
=========================================================
QUESTION 9
Display department and number of employees
=========================================================
*/

SELECT d.dept_name, COUNT(e.emp_id) AS total_employees
FROM departments d
LEFT JOIN employees e
ON d.dept_id = e.dept_id
GROUP BY d.dept_name;



/*
=========================================================
QUESTION 10
Display employees earning more than their managers
=========================================================
*/

SELECT e.name AS Employee, e.salary, m.name AS Manager, m.salary
FROM employees e
JOIN employees m
ON e.manager_id = m.emp_id
WHERE e.salary > m.salary;



/*
=========================================================
QUESTION 11
Display all combinations of employees and departments
CROSS JOIN
=========================================================
*/

SELECT e.name, d.dept_name
FROM employees e
CROSS JOIN departments d;



/*
=========================================================
QUESTION 12
Find departments with no employees
=========================================================
*/

SELECT d.dept_name
FROM departments d
LEFT JOIN employees e
ON d.dept_id = e.dept_id
WHERE e.emp_id IS NULL;



/*
=========================================================
QUESTION 13
Find employees whose department is Finance
=========================================================
*/

SELECT e.name
FROM employees e
JOIN departments d
ON e.dept_id = d.dept_id
WHERE d.dept_name = 'Finance';



/*
=========================================================
QUESTION 14
Display employees sorted by department name
=========================================================
*/

SELECT e.name, d.dept_name
FROM employees e
JOIN departments d
ON e.dept_id = d.dept_id
ORDER BY d.dept_name;



/*
=========================================================
QUESTION 15
Find highest paid employee in each department
=========================================================
*/

SELECT d.dept_name, MAX(e.salary) AS highest_salary
FROM employees e
JOIN departments d
ON e.dept_id = d.dept_id
GROUP BY d.dept_name;
