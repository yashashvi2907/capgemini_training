/*
=========================================================
QUESTION 11
Find departments where average salary is greater than 65000
=========================================================
*/

SELECT department, AVG(salary) AS avg_salary
FROM employees
GROUP BY department
HAVING AVG(salary) > 65000;



/*
=========================================================
QUESTION 12
Find departments with more than 2 employees
=========================================================
*/

SELECT department, COUNT(*) AS employee_count
FROM employees
GROUP BY department
HAVING COUNT(*) > 2;



/*
=========================================================
QUESTION 13
Find cities where more than one employee lives
=========================================================
*/

SELECT city, COUNT(*) AS total_people
FROM employees
GROUP BY city
HAVING COUNT(*) > 1;



/*
=========================================================
QUESTION 14
Find department having maximum average salary
=========================================================
*/

SELECT department, AVG(salary) AS avg_salary
FROM employees
GROUP BY department
ORDER BY avg_salary DESC
LIMIT 1;



/*
=========================================================
QUESTION 15
Find department where total salary expenditure is highest
=========================================================
*/

SELECT department, SUM(salary) AS total_salary
FROM employees
GROUP BY department
ORDER BY total_salary DESC
LIMIT 1;
